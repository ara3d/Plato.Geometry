
// A length measurement 
type Length
    implements IMeasure
{
    Meters: float;
}

type Mass
    implements IMeasure
{
    Kilograms: float;
}

type Temperature
    implements IMeasure
{
    Celsius: float;
}


// Unix date-time representation: float of non-leap seconds elapsed since January 1st, 1970.
// https://en.wikipedia.org/wiki/Unix_time
type DateTime
    implements ICoordinate
{
    Value: float; 
}


// A fraction of two integer
type Rational 
	implements IValue
{
	Numerator: Integer;
	Denominator: Integer;
}

// An arbitrary fraction 
type Fraction
    implements IValue
{
    Numerator: float;
    Denominator: float;
}


// A value in the range of 0 to 1 that represents the probability of an occurence
type Probability
    implements IMeasure
{
    Value: float;
}

// A two dimensional value, have has a real and imaginary component. 
type Complex
    implements IVector
{
    Real: float;
    Imaginary: float;
}
